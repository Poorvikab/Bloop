## ğŸ® Game Overview

**Timeline Detective** is an immersive historical mystery game where players solve crimes and mysteries by correctly ordering historical events, analyzing cause-and-effect relationships, and piecing together timelines from scattered clues. Players become time-traveling detectives who must restore corrupted historical timelines to solve cases across different eras.

- Master chronological thinking and historical sequencing
- Understand cause-and-effect relationships in history
- Learn historical facts through engaging narratives
- Develop critical thinking and deductive reasoning
- Connect events across different time periods and cultures

## ğŸ² Game Mechanics

### Core Gameplay

**The Mystery**:
- Each case presents a "corrupted timeline" with events out of order
- Players receive clues (documents, artifacts, testimonies)
- Must arrange events in correct chronological order
- Solve the mystery by understanding what really happened

**Timeline Mechanics**:

1. **Event Cards**: Each card represents a historical event
   - Date (hidden initially)
   - Description
   - Location
   - Key figures
   - Consequences

2. **Clue System**:
   - **Primary Sources**: Letters, diaries, official documents
   - **Physical Evidence**: Artifacts, photographs, maps
   - **Testimonies**: Witness accounts (may be unreliable)
   - **Context Clues**: Cultural, technological, political context

3. **Deduction Board**:
   - Place events on timeline
   - Draw connections between events
   - Mark cause-and-effect relationships
   - Identify contradictions

4. **Verification**:
   - Submit timeline for verification
   - Get feedback on incorrect placements
   - Earn points for correct sequences
   - Unlock new clues for mistakes

  hints: string[];

interface Case {
  id: string;
  title: string;
  description: string;
  era: string;
  difficulty: number;
  events: HistoricalEvent[];
  clues: Clue[];
  solution: string[]; // Correct order of event IDs

}

class TimelineGame {
  private currentCase: Case;
  private playerTimeline: HistoricalEvent[];
  private unlockedClues: Clue[];
  
  validateTimeline(): ValidationResult {
    const correct = [];
    const incorrect = [];
    
    for (let i = 0; i < this.playerTimeline.length; i++) {
      const event = this.playerTimeline[i];
      const correctIndex = this.currentCase.solution.indexOf(event.id);
      
      if (correctIndex === i) {
        correct.push(event);
      } else {
        incorrect.push({
          event,
          correctPosition: correctIndex,
          currentPosition: i
        });
      }
    }
    
    return {
      correct,
      incorrect,
      score: (correct.length / this.currentCase.events.length) * 100
    };
  }
  
  analyzeClue(clue: Clue): ClueAnalysis {
    // Use NLP to extract date hints
    const dateHints = this.extractDateHints(clue.content);
    
    // Find related events
    const relatedEvents = this.currentCase.events.filter(e =>
      clue.relatedEvents.includes(e.id)
    );
    
    // Determine reliability
    const trustScore = this.calculateTrustScore(clue);
    
    return {
      dateHints,
      relatedEvents,
      trustScore,
      suggestions: this.generateSuggestions(clue, relatedEvents)
    };
  }
}

        relationships = self.db.get_relationships()

        self,

        difficulty: int,

        """Generate a mystery case"""

class CaseValidator:
    def validate_timeline(
        self,
        submitted: List[str],
        correct: List[str]
    ) -> Dict:
        """Validate player's timeline submission"""
        
        # Calculate accuracy
        correct_positions = sum(
            1 for i, event_id in enumerate(submitted)
            if event_id == correct[i]
        )
        
        accuracy = correct_positions / len(correct)
        
        # Find mistakes
        mistakes = []
        for i, event_id in enumerate(submitted):
            if event_id != correct[i]:
                correct_pos = correct.index(event_id)
                mistakes.append({
                    'event_id': event_id,
                    'submitted_position': i,
                    'correct_position': correct_pos,
                    'hint': self.generate_hint(event_id, correct_pos)
                })
        
        return {
            'accuracy': accuracy,
            'correct_positions': correct_positions,
            'mistakes': mistakes,
            'score': self.calculate_score(accuracy, len(mistakes))
        }
